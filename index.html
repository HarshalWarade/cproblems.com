Q7] Write C program to check given number is prime or not. 


 #include <stdio.h>
 #include <conio.h>
 int main()
 {
    int i = 0, n, d = 0;
    printf("Please give input a number: \n");
    scanf("%d", &n);
    for (i = 2; i <= (n / 2); i++)
    {
        if (n % i == 0)
        {
            d = 1;
            break;
        }
    }
    if (d == 1)
        printf("given number is not a prime number\n");
    else
        printf("given number is a prime number\n");
    getch();
    return 0;
 }
